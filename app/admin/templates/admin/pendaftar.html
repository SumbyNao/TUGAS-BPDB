class User(UserMixin, db.Model): id = db.Column(db.Integer, primary_key=True) nama_lengkap = db.Column(db.String(100)) email = db.Column(db.String(120), unique=True) password_hash = db.Column(db.String(128)) role = db.Column(db.String(20), default='user') # 'admin' atau 'user' pendaftar = db.relationship('Pendaftar', backref='user', uselist=False)
